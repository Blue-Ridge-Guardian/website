Secure Filtering Examples

// Pre-query sanitizer
function sanitizeInput(prompt) {
  return prompt.replace(/secret|password/gi, '[REDACTED]');
}

// Post-query output filter
function sanitizeOutput(response) {
  if (response.includes("confidential")) return "[REDACTED OUTPUT]";
  return response;
}
